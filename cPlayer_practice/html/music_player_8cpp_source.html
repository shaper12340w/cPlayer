39void ExecuteMCI(MCIDEVICEID id, UINT msg, DWORD_PTR command, DWORD_PTR param) {
40 DWORD result = mciSendCommandW(id, msg, command, param);
41 if (result != 0) {
42 WCHAR reason[256];
43 mciGetErrorStringW(result, reason, 256);
44 switch (msg) {
45 case MCI_OPEN:
46 if (result == MCIERR_INTERNAL) {
47 wchar_t tempPath[MAX_PATH] = L"";
48 SaveTemporaryMP3(getStringArrayValue(&player.pathArray, player.playIndex), tempPath, MAX_PATH);
49 changeStringArrayValue(&player.pathArray, tempPath, player.playIndex);
50 addIntArrayValue(&player.deletedArray, player.playIndex);
51 uprintf(6,L"Warning : MP3 tag error\nPlay with temp File..");
52 Ready();
53 return;
54 }
55 else
56 uprintf(6,L"Error Opening Music : \n%ls\nCode : %d", reason,result);
57 break;
58 case MCI_PLAY: uprintf(6,L"Error Playing Music : \n%ls\nCode : %d", reason, result); break;
59 case MCI_PAUSE: uprintf(6,L"Error Pausing Music : \n%ls\nCode : %d", reason, result);break;
60 case MCI_STATUS:uprintf(6,L"Error Getting Information of Music : \n%ls\nCode : %d", reason, result);break;
61 case MCI_STOP: uprintf(6,L"Error Stoping Music : \n%ls\nCode : %d", reason, result);break;
62 case MCI_CLOSE: uprintf(6,L"Error Closing Music : \n%ls\nCode : %d", reason, result);break;
63 default:
64 uprintf(6,L"Player Error : \n%ls\nCode : %d", reason, result);
65 break;
66 }
67 exit(1);
68 }
